340  template <typename InputIterator>
341  const_iterator insert(InputIterator first, InputIterator last, value_type value = 0)
342  {
343  assert(value <= std::numeric_limits<int32_t>::max());
344 
345  if (first == last) return cend();
346 
347  index_type parent = 1;
348 
349  // 登録済み遷移をスキップする
350  for (index_type idx = at(parent, static_cast<std::uint8_t>(*first)); idx != 0; idx = at(parent, static_cast<std::uint8_t>(*first)))
351  {
352  parent = idx;
353  ++first;
354  if (first == last) break;
355  }
356 
357  if (!has_null(parent) || first != last)
358  {
359  // 新規文字列による遷移を追加
360  while (first != last)
361  {
362  std::uint16_t label = static_cast<std::uint8_t>(*first++);
363  index_type base = add(parent, label);
364  parent = base + label;
365  }
366  assert(parent != 1);
367 
368  // 終端の空遷移を追加
369  index_type idx = add(parent, null_value) + null_value;
370  (m_c.data() + idx)->m_base = -value;
371 
372  ++m_c.front().m_base;
373  }
374 
375  return const_iterator(m_c, parent);
376  }
