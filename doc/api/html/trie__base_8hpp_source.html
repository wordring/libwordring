273  template <typename InputIterator>
274  const_iterator insert(InputIterator first, InputIterator last, value_type value = 0)
275  {
276  assert(value <= std::numeric_limits<int32_t>::max());
277 
278  if (first == last) return cend();
279 
280  index_type parent = 1;
281 
282  // 登録済み遷移をスキップする
283  for (index_type idx = at(parent, static_cast<std::uint8_t>(*first)); idx != 0; idx = at(parent, static_cast<std::uint8_t>(*first)))
284  {
285  parent = idx;
286  ++first;
287  if (first == last) break;
288  }
289  if (first != last) ++m_c.front().m_base;
290 
291  // 空遷移の解決
292  if (has_child(parent)) // 既存遷移有り、新規文字列無し
293  {
294  if (first == last && !has_null(parent))
295  {
296  index_type base = add(parent, null_value);
297  assert((m_c.data() + base + null_value)->m_base == 0);
298  (m_c.data() + base + null_value)->m_base = -value;
299  ++m_c.front().m_base;
300  }
301  }
302  else if (parent != 1 && !has_null(parent) && first != last) // 後続遷移無し、新規文字列有り
303  {
304  index_type val = (m_c.data() + parent)->m_base;
305  assert(val <= 0);
306  index_type base = add(parent, null_value);
307  assert((m_c.data() + base + null_value)->m_base == 0);
308  (m_c.data() + base + null_value)->m_base = val;
309  }
310 
311  // 新規文字列による遷移を追加
312  index_type base = 0;
313  while (first != last)
314  {
315  std::uint16_t label = static_cast<std::uint8_t>(*first);
316  ++first;
317  base = add(parent, label);
318  parent = base + label;
319  }
320  if (base != 0)
321  {
322  assert((m_c.data() + parent)->m_base == 0);
323  (m_c.data() + parent)->m_base = -value;
324  }
325 
326  assert(parent != 1);
327 
328  return const_iterator(m_c, parent);
329  }

342  {

344  assert(0 <= idx && idx < limit());
