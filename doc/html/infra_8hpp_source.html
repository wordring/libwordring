97  {
98  assert(first != last);
99  unsigned char ch1 = static_cast<unsigned char>(*first++);

101 
102  assert(first != last);
103  unsigned char ch2 = static_cast<unsigned char>(*first++);
104  assert((ch2 & 0xC0u) == 0x80u);

106 
107  assert(first != last);
108  unsigned char ch3 = static_cast<unsigned char>(*first++);
109  assert((ch3 & 0xC0u) == 0x80u);
110  if ((ch1 & 0xF0u) == 0xE0u) return static_cast<uint32_t>(((ch1 & 0xFu) << 12) + ((ch2 & 0x3Fu) << 6) + (ch3 & 0x3Fu));
111 
112  assert(first != last);
113  unsigned char ch4 = static_cast<unsigned char>(*first++);
114  assert((ch4 & 0xC0u) == 0x80u);
115  if ((ch1 & 0xF8u) == 0xF0u) return static_cast<uint32_t>(((ch1 & 0x7u) << 18) + ((ch2 & 0x3Fu) << 12) + ((ch3 & 0x3Fu) << 6) + (ch4 & 0x3Fu));

153  /*!
154  @brief ユニコード・コードポイントから文字列に変換する。
155 
156  @tparam OutputIterator 出力イテレータの型。
157 
158  @param [in] ch ユニコード・コードポイント。
159  @param [out] output 出力先となる出力イテレータ。
160 
161  この関数はライブラリ内部での利用を意図する。
162  入力にエラーがある場合を考慮しない。
163 
164  OutputIteratorのvalue_typeの型に応じて文字符号化が決定される。
165  std::back_insert_iterator<>が与えられた場合、container_type::value_typeを参照する。
166  - char8_t/charの場合、utf-8。
167  - char16_tの場合、utf-16。
168  - char32_t/scalar_valueの場合、utf-32。
169 
170  utf-8文字符号化は、処理系がchar8_tをサポートする場合char8_t、しない場合charで有効化される。
171  */
172  template <typename OutputIterator,
173  typename std::enable_if_t<
174  std::conjunction_v<
175  is_iterator<OutputIterator>,
176  std::disjunction<
177  std::is_same<typename std::iterator_traits<OutputIterator>::value_type, char_utf8_type>,
178  std::is_same<typename OutputIterator::container_type::value_type, char_utf8_type>>>, nullptr_t> = nullptr>
179  inline void to_string(uint32_t ch, OutputIterator output)
180  {
181  if ((ch & 0xFFFFFF80u) == 0) *output++ = static_cast<unsigned char>(ch); // 一バイト文字
182  else if ((ch & 0xFFFFF800u) == 0) // 二バイト文字
183  {
184  *output++ = static_cast<unsigned char>(0xC0u + (ch >> 6));
185  *output++ = static_cast<unsigned char>(0x80u + (ch & 0x3Fu));
186  }
187  else if ((ch & 0xFFFF0000u) == 0) // 三バイト文字
188  {
189  *output++ = static_cast<unsigned char>(0xE0u + (ch >> 12));
190  *output++ = static_cast<unsigned char>(0x80u + ((ch & 0xFC0u) >> 6));
191  *output++ = static_cast<unsigned char>(0x80u + (ch & 0x3Fu));
192  }
193  else if ((ch & 0xFFE00000u) == 0) // 四バイト文字
194  {
195  *output++ = static_cast<unsigned char>(0xF0u + (ch >> 18));
196  *output++ = static_cast<unsigned char>(0x80u + ((ch & 0x3F000u) >> 12));
197  *output++ = static_cast<unsigned char>(0x80u + ((ch & 0xFC0u) >> 6));
198  *output++ = static_cast<unsigned char>(0x80u + (ch & 0x3Fu));
199  }
200  else assert(false);
201  }

274  /*! 小文字化
275 
276  @tparam InputIterator value_typeがcharである入力イテレータ。
277  @tparam OutputIterator 出力イテレータ。
278 
279  @param[in] first 入力（半開区間）の先頭を示すイテレータ。
280  @param[in] last 入力（半開区間）の終端を示すイテレータ。
281  @param[out] output 出力先を示すイテレータ。

292  static_assert(std::is_same_v<typename std::iterator_traits<InputIterator>::value_type, char>);
293 
294  while (first != last)
295  {
296  unsigned char ch = *first++;
297  if (0x41u <= ch && ch <= 0x5Au) ch += 0x20u;
298  *output++ = ch;
299  }
300  }
