{

{
return val.type() == typeid(simple_block const);
}
inline bool is_preserved_token(std::any const& val)
{
return is_css_token(val)
&& !( is_function_token(val)
|| is_open_curly_token(val)
|| is_open_paren_token(val)
|| is_open_square_token(val));
}
inline bool is_component_value(std::any const& val)
{
return is_preserved_token(val) || is_function(val) || is_simple_block(val);
}

template <typename ErrorHandler = std::nullptr_t>
inline std::vector<std::any> normalize_into_token_stream(std::u32string&& in, ErrorHandler handler = nullptr)
{
std::vector<std::any> out;
in.erase(filter_code_points(in.begin(), in.end()), in.end());

return out;
}

// 5.4. Parser Algorithms
//
// https://drafts.csswg.org/css-syntax-3/#parser-algorithms
// https://triple-underscore.github.io/css-syntax-ja.html#parser-algorithms
// --------------------------------------------------------------------------------------------
// --------------------------------------------------------------------------------------------
// 5.4.1. Consume a list of rules
//
// https://drafts.csswg.org/css-syntax-3/#consume-list-of-rules
// https://triple-underscore.github.io/css-syntax-ja.html#consume-list-of-rules
// --------------------------------------------------------------------------------------------
template <typename BidirectionalIterator, typename ErrorHandler>
inline std::vector<css_component> consume_list_of_rules(
BidirectionalIterator first, BidirectionalIterator last, bool top_level = false, ErrorHandler handler = nullptr)
{
std::vector<css_component> rules;
while (first != last)
{
std::any tkn = *first++;
if (is_whitespace_token(tkn)) continue;
else if (is_CDC_token(tkn) || is_CDC_token(tkn))
{
if (top_level) continue;
else
{
--first;
auto [it, rule] = consume_qualified_rule(first, last, handler);
first = it;
if (rule.has_value()) rules.push_back(*rule);
}
}
else if (is_at_keyword_token(tkn))
{
--first;
auto [it, rule] = consume_at_rule(first, last, handler);
first = it;
rules.push_back(rule);
}
else if (is_eof_token(tkn))
{
assert(false);
break;
}
else
{
--first;
auto [it, rule] = consume_qualified_rule(first, last, handler);
first = it;
if (rule.has_value()) rules.push_back(*rule);
}
}
return rules;
}
