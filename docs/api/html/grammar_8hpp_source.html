668  template <typename NodePointer>
669  bool match(NodePointer const& np, match_context<NodePointer> const& ctx) const
670  {
671  using traits = typename wordring::html::node_traits<NodePointer>;
672  using wordring::whatwg::is_ascii_case_insensitive_match;
673  using wordring::whatwg::split_on_ascii_whitespace;
674  using wordring::encoding_cast;
675  using wordring::whatwg::to_ascii_lowercase;
676 
677  assert(!(ctx.m_type_name == document_type_name::Xml
678  && (ctx.m_mode_name == document_mode_name::LimitedQuirks || ctx.m_mode_name == document_mode_name::Quirks)));
679 
680  // 要素以外のノードに属性は無い
681  if (!traits::is_element(np)) return false;
682 
683  // 属性名照合
684  auto it1 = traits::abegin(np);
685  auto it2 = traits::aend(np);
686  while (it1 != it2)
687  {
688  // 名前空間接頭辞
689  if (m_name.prefix())
690  {
691  if (ctx.m_namespace_enabled)
692  {
693  std::u32string const& prefix1 = m_name.prefix()->string();
694  std::u32string uri2 = encoding_cast<std::u32string>(traits::get_namespace(*it1));
695 
696  if (prefix1 == U"*"); // 接頭辞を持たない属性を含めて全てにマッチ
697  else if (prefix1 == U"" && !uri2.empty())
698  {
699  ++it1;
700  continue;
701  }
702  else
703  {
704  std::u32string const* uri1 = ctx.get_namespace(prefix1);
705  if (uri1 == nullptr || *uri1 != uri2)
706  {
707  ++it1;
708  continue;
709  }
710  }
711  }
712  else
713  {
714  // 名前空間をサポートしないにもかかわらず、セレクタで名前空間が指定された。
715  // TOTO: ワイルドカード、空の接頭辞も無効なセレクタとして何にもマッチしないことにしているが、
716  // どうすべきか？
717  ++it1;
718  continue;
719  }
720  }
721  else
722  {
723  if (!traits::get_namespace(*it1).empty())
724  {
725  ++it1;
726  continue;
727  }
728  }
729 
730  // 属性名
731  std::u32string const& n1 = m_name.name();
732  std::u32string n2 = encoding_cast<std::u32string>(traits::get_local_name(*it1));
733  // XML はASCII大小区別、HTMLは大小無視
734  if (ctx.m_type_name == document_type_name::Xml)
735  {
736  if (n1 == n2) break;
737  }
738  else
739  {
740  if (is_ascii_case_insensitive_match(n1.begin(), n1.end(), n2.begin(), n2.end())) break;
741  }
742 
743  ++it1;
744  }
745  if (it1 == it2) return false;
746 
747  // 属性値照合
748  switch (m_matcher)
749  {
750  case U'\0':// [attr]
751  assert(m_value.empty());
752  return true;
753  case U'=': // [attr=val]
754  {
755  std::u32string s = encoding_cast<std::u32string>(traits::value(*it1));
756  if (m_modifier == U'i') return is_ascii_case_insensitive_match(m_value.begin(), m_value.end(), s.begin(), s.end());
757  else return m_value == s;
758  }
759  case U'~': // [attr~=val]
760  {
761  std::u32string tmp = encoding_cast<std::u32string>(traits::value(*it1));
762  std::vector<std::u32string> v;
763  split_on_ascii_whitespace(tmp.begin(), tmp.end(), std::back_inserter(v));
764  for (std::u32string const& s : v)
765  {
766  if (m_modifier == U'i' && is_ascii_case_insensitive_match(m_value.begin(), m_value.end(), s.begin(), s.end())) return true;
767  else if(m_value == s) return true;
768  }
769  break;
770  }
771  case U'|': // [attr|=val]
772  {
773  std::u32string s = encoding_cast<std::u32string>(traits::value(*it1));
774  if (m_modifier == U'i')
775  {
776  if (m_value.size() < s.size()
777  && is_ascii_case_insensitive_match(m_value.begin(), m_value.end(), s.begin(), s.begin() + m_value.size())
778  && s[m_value.size()] == U'-') return true;
779  }
780  else
781  {
782  if (m_value.size() < s.size()
783  && std::equal(m_value.begin(), m_value.end(), s.begin(), s.begin() + m_value.size())
784  && s[m_value.size()] == U'-') return true;
785  }
786  break;
787  }
788  case U'^': // [attr^=val]
789  {
790  std::u32string s = encoding_cast<std::u32string>(traits::value(*it1));
791  if (m_modifier == U'i')
792  {
793  if (m_value.size() <= s.size()
794  && is_ascii_case_insensitive_match(m_value.begin(), m_value.end(), s.begin(), s.begin() + m_value.size())) return true;
795  }
796  else
797  {
798  if (m_value.size() <= s.size()
799  && std::equal(m_value.begin(), m_value.end(), s.begin(), s.begin() + m_value.size())) return true;
800  }
801  break;
802  }
803  case U'$': // [attr$=val]
804  {
805  std::u32string s = encoding_cast<std::u32string>(traits::value(*it1));
806  if (s.size() < m_value.size()) return false;
807  if (m_modifier == U'i')
808  {
809  if (is_ascii_case_insensitive_match(m_value.rbegin(), m_value.rend(), s.rbegin(), s.rbegin() + m_value.size())) return true;
810  }
811  else
812  {
813  if (std::equal(m_value.rbegin(), m_value.rend(), s.rbegin(), s.rbegin() + m_value.size())) return true;
814  }
815  break;
816  }
817  case U'*': // [attr*=val]
818  {
819  std::u32string s1 = encoding_cast<std::u32string>(traits::value(*it1));
820  std::u32string s2 = m_value;
821  if (m_modifier == U'i')
822  {
823  std::u32string tmp;
824  to_ascii_lowercase(s1.begin(), s1.end(), std::back_inserter(tmp));
825  std::swap(s1, tmp);
826  tmp.clear();
827  to_ascii_lowercase(s2.begin(), s2.end(), std::back_inserter(tmp));
828  s2 = std::move(tmp);
829  }
830  if (s1.find(s2) != std::u32string::npos) return true;
831  break;
832  }
833  default:
834  break;
835  }
836  return false;
837  }
838 
839  private:
841  wq_name m_name;
842  char32_t m_matcher;
843  std::u32string m_value;
844 
851  char32_t m_modifier;
852  };
