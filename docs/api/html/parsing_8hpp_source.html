440  template <typename ErrorHandler>
441  inline std::optional<syntax_primitive> parse_rule(token_stream& in, ErrorHandler handler)
442  {
443  while (in.next_input_token().type() == syntax_primitive_name::WhitespaceToken) in.consume();
444  if (in.next_input_token().type() == syntax_primitive_name::EofToken) return std::optional<syntax_primitive>();
445 
446  syntax_primitive rule;
447  if (in.next_input_token().type() == syntax_primitive_name::AtKeywordToken)
448  {
449  rule = consume_at_rule(in, handler);
450  }
451  else
452  {
453  std::optional<qualified_rule> q = consume_qualified_rule(in, handler);
454  if (!q) return std::optional<syntax_primitive>();
455  rule = std::move(*q);
456  }
457 
458  while (in.next_input_token().type() == syntax_primitive_name::WhitespaceToken) in.consume();
459  if (in.next_input_token().type() == syntax_primitive_name::EofToken)
460  {
461  return std::make_optional<syntax_primitive>(std::move(rule));
462  }
463 
464  return std::optional<syntax_primitive>();
465  }

823  //
824  // https://drafts.csswg.org/css-syntax-3/#consume-list-of-rules

841  template <typename ErrorHandler>
842  inline std::vector<syntax_primitive>
843  consume_list_of_rules(token_stream& in, bool top_level, ErrorHandler handler)
844 
845  {
846  std::vector<syntax_primitive> rules;
847 
848  while (true)
849  {
850  syntax_primitive const& tkn = in.consume();
851 
852  switch (tkn.type())
853  {
854  case syntax_primitive_name::WhitespaceToken:
855  continue;
856  case syntax_primitive_name::EofToken:
857  return rules;
858  case syntax_primitive_name::CdoToken:
859  case syntax_primitive_name::CdcToken:
860  if (top_level)
861  {
862  continue;
863  }
864  else
865  {
866  in.reconsume();
867  std::optional<qualified_rule> rule = consume_qualified_rule(in, handler);
868  if (rule.has_value()) rules.push_back(std::move(*rule));
869  }
870  case syntax_primitive_name::AtKeywordToken:
871  in.reconsume();
872  rules.push_back(consume_at_rule(in, handler));
873  continue;
874  default:
875  in.reconsume();
876  std::optional<qualified_rule> rule = consume_qualified_rule(in, handler);
877  if (rule.has_value()) rules.push_back(std::move(*rule));
878  }
879  }
880 
881  assert(false);
882  return rules;
883  }

908  template <typename ErrorHandler>
909  inline at_rule consume_at_rule(token_stream& in, ErrorHandler handler)
910  {
911  syntax_primitive const& tkn = in.consume();
912  assert(in.current_input_token().type() == syntax_primitive_name::AtKeywordToken);
913 
914  at_rule rule;
915  rule.m_name = tkn.get<at_keyword_token>().m_value;
916 
917  while (true)
918  {
919  syntax_primitive const& tkn = in.consume();
920  switch (tkn.type())
921  {
922  case syntax_primitive_name::SemicolonToken:
923  return rule;
924  case syntax_primitive_name::EofToken:
925  if constexpr (std::is_invocable_v<ErrorHandler, token_stream&>) handler(in);
926  return rule;
927  case syntax_primitive_name::OpenCurlyToken:
928  rule.m_block = std::make_unique<simple_block>(consume_simple_block(in, handler));
929  return rule;
930  case syntax_primitive_name::SimpleBlock:
931  if (tkn.get<simple_block>().m_associated_token == U'{')
932  {
933  rule.m_block = std::make_unique<simple_block>(tkn.get<simple_block>());
934  return rule;
935  }
936  default:
937  in.reconsume();
938  rule.m_prelude.push_back(consume_component_value(in, handler));
939  }
940  }
941 
942  assert(false);
943  return at_rule();
944  }

967  template <typename ErrorHandler>
968  inline std::optional<qualified_rule> consume_qualified_rule(token_stream& in, ErrorHandler handler)
969  {
970  qualified_rule rule;
971 
972  while (true)
973  {
974  syntax_primitive const& tkn = in.consume();
975  switch (tkn.type())
976  {
977  case syntax_primitive_name::EofToken:
978  if constexpr (std::is_invocable_v<ErrorHandler, token_stream&>) handler(in);
979  return std::optional<qualified_rule>();
980  case syntax_primitive_name::OpenCurlyToken:
981  rule.m_block = consume_simple_block(in, handler);
982  return std::make_optional(std::move(rule));
983  case syntax_primitive_name::SimpleBlock:
984  if (tkn.get<simple_block>().m_associated_token == U'{')
985  {
986  rule.m_block = tkn.get<simple_block>();
987  return std::make_optional(std::move(rule));
988  }
989  default:
990  in.reconsume();
991  rule.m_prelude.push_back(consume_component_value(in, handler));
992  }
993  }
994 
995  assert(false);
996  return std::optional<qualified_rule>();
