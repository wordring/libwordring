284  {

288  css_token const& current_input_token()
289  {
290  return m_it != m_v.cend() ? *m_it : m_eof;

292 
293  css_token const& next_input_token()
294  {
295  if (m_it == m_v.cend()) return m_eof;
296 
297  const_iterator it = std::next(m_it);
298 
299  return it != m_v.cend() ? *it : m_eof;
300  }
301 
302  css_token const& consume()
303  {
304  if (m_reconsume_flag) m_reconsume_flag = false;
305  else if (m_it != m_v.cend()) ++m_it;
306 
307  return current_input_token();
308  }
309 
310  void reconsume()
311  {
312  assert(m_reconsume_flag == false);
313 
314  m_reconsume_flag = true;
315  }
316 
317  private:

680  template <typename ErrorHandler>
681  inline parse_result<css_component> parse_rule(token_stream& in, ErrorHandler handler)
682  {
683  using result_type = parse_result<css_component>;
684 
685  while (is_whitespace_token(in.next_input_token())) in.consume();
686  if (is_eof_token(in.next_input_token())) return result_type();
687 
688  css_component rule;
689  if (is_at_keyword_token(in.next_input_token()))
690  {
691  rule = consume_at_rule(in, handler);
692  }
693  else
694  {
695  std::optional<qualified_rule> q = consume_qualified_rule(in, handler);
696  if (!q) return result_type();
697  rule = std::move(*q);
698  }
699 
700  while (is_whitespace_token(in.next_input_token())) in.consume();
701  if (is_eof_token(in.next_input_token())) return result_type(std::move(rule));
702 
703  return result_type();
704  }
