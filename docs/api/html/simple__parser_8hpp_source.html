265  void parse(iterator first, iterator last)
266  {
267  if constexpr (sizeof(*first) == 4)
268  {
269  while (first != last)
270  {
271  push_back(*first);
272  ++first;
273  }
274  }
275  else if constexpr (sizeof(*first) == 2)
276  {
277  std::u32string s;
278  encoding_cast(first, last, std::back_inserter(s));
279  for (char32_t cp : s) push_back(cp);
280  }
281  else if constexpr (sizeof(*first) == 1)
282  {
283  using namespace wordring::encoding;
284 
285  decoder dec(static_cast<encoding_name>(0), static_cast<error_mode_name>(0));
286  std::deque<uint32_t> buffer;
287  auto out = std::back_inserter(buffer);
288  iterator it1, it2;
289  result_value rv;
290 
291  Start:
292  dec = decoder(base_type::m_encoding_name, error_mode_name::Replacement);
293  it1 = first;
294  it2 = last;
295  while (it1 != it2)
296  {
297  buffer.clear();
298  rv = dec.push(*it1++, out);
299  if (buffer.empty()) continue;
300 
301  for (uint32_t cp : buffer)
302  {
303  base_type::push_code_point(cp);
304  if (m_updated_encoding_name != static_cast<encoding_name>(0))
305  {
306  clear(base_type::m_encoding_confidence, m_updated_encoding_name);
307  goto Start;
308  }
309  }
310  }
311  }
312  else assert(false);
313  }
