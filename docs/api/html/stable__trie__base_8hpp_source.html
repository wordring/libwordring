447  template <typename InputIterator>
448  const_iterator insert(InputIterator first, InputIterator last, value_type value = 0)
449  {
450  assert(value <= std::numeric_limits<int32_t>::max());
451 
452  if (first == last) return cend();
453 
454  index_type parent = 1;
455 
456  // 登録済み遷移をスキップする
457  for (index_type idx = at(parent, static_cast<std::uint8_t>(*first)); idx != 0; idx = at(parent, static_cast<std::uint8_t>(*first)))
458  {
459  parent = idx;
460  ++first;
461  if (first == last) break;
462  }
463 
464  if (!has_null(parent) || first != last)
465  {
466  // 新規文字列による遷移を追加
467  while (first != last)
468  {
469  std::uint16_t label = static_cast<std::uint8_t>(*first++);
470  index_type base = add(parent, label);
471  parent = base + label;
472  }
473  assert(parent != 1);
474 
475  // 終端の空遷移を追加
476  index_type idx = add(parent, null_value) + null_value;
477  (m_c.data() + idx)->m_base = -value;
478 
479  ++m_c.front().m_base;
480  }
481 
482  return const_iterator(m_c, parent);
483  }
