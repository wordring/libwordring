586  {
587  index_type parent = pos.m_parent;

592  index_type idx = allocate(std::move(value));
593 
594  node_type* d = m_c.data();
595 
596  index_type child = (d + parent)->m_child;
597  index_type after = pos.m_index;
598  index_type before = 0;
599  if (after != 0) before = (child != after) ? (d + after)->m_prev : 0;
600  else before = (child != 0) ? (d + child)->m_prev : 0;
601 
602  if (before == 0)
603  {
604  (d + parent)->m_child = idx;

606  (d + idx)->m_prev = tail;

611  (d + idx)->m_prev = before;
