433  template <typename InputIterator>
434  const_iterator insert(InputIterator first, InputIterator last, value_type value = 0)
435  {
436  assert(value <= static_cast<value_type>(std::numeric_limits<int32_t>::max()));
437 
438  if (first == last) return cend();
439 
440  index_type parent = 1;
441 
442  // 登録済み遷移をスキップする
443  for (index_type idx = at(parent, static_cast<std::uint8_t>(*first)); idx != 0; idx = at(parent, static_cast<std::uint8_t>(*first)))
444  {
445  parent = idx;
446  ++first;
447  if (first == last) break;
448  }
449  if (first != last) ++m_c.front().m_base;
450 
451  // 空遷移の解決
452  if (has_child(parent)) // 既存遷移有り、新規文字列無し
453  {
454  if (first == last && !has_null(parent))
455  {
456  index_type base = add(parent, null_value);
457  assert((m_c.data() + base + null_value)->m_base == 0);
458  (m_c.data() + base + null_value)->m_base = -static_cast<index_type>(value);
459  ++m_c.front().m_base;
460  }
461  }
462  else if (parent != 1 && !has_null(parent) && first != last) // 後続遷移無し、新規文字列有り
463  {
464  index_type val = (m_c.data() + parent)->m_base;
465  assert(val <= 0);
466  index_type base = add(parent, null_value);
467  assert((m_c.data() + base + null_value)->m_base == 0);
468  (m_c.data() + base + null_value)->m_base = val;
469  }
470 
471  // 新規文字列による遷移を追加
472  index_type base = 0;
473  while (first != last)
474  {
475  std::uint16_t label = static_cast<std::uint8_t>(*first);
476  ++first;
477  base = add(parent, label);
478  parent = base + label;
479  }
480  if (base != 0)
481  {
482  assert((m_c.data() + parent)->m_base == 0);
483  (m_c.data() + parent)->m_base = -static_cast<index_type>(value);
484  }
485 
486  assert(parent != 1);
487 
488  return const_iterator(m_c, parent);
489  }
