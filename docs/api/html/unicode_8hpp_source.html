139  {
140  if ((ch & 0xFFFFFF80u) == 0) *output++ = static_cast<unsigned char>(ch); // 一バイト文字
141  else if ((ch & 0xFFFFF800u) == 0) // 二バイト文字
142  {
143  *output++ = static_cast<unsigned char>(0xC0u + (ch >> 6));
144  *output++ = static_cast<unsigned char>(0x80u + (ch & 0x3Fu));

146  else if ((ch & 0xFFFF0000u) == 0) // 三バイト文字
147  {
