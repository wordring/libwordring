93  inline void percent_decode(InputIterator first, InputIterator last, OutputIterator output)
94  {
95  static_assert(sizeof(*first) == 1);
96 
97  while (first != last)
98  {
99  if (starts_with_percent_encoded_byte(first, last))

107  continue;
108  }

110  }

275  inline std::string percent_encode_after_encoding(
276  encoding_name name, std::u32string const& in, PercentEncodeSet set, bool spaceAsPlus = false)
277  {
278  using namespace wordring::whatwg::encoding;
279  // 1.
280  coder c = get_encoder(name);
281  // 2.
282  io_queue<char32_t> inputQueue = to_io_queue_convert(in.begin(), in.end());
283  // 3.
284  std::string output;
285  // 4.
286  result_value potentialError = result_error{};
287  // 5.
288  while (std::holds_alternative<result_error>(potentialError))
289  {
290  // 5.1.
291  io_queue<char> encodeOutput;
292  // 5.2.
293  potentialError = encode_or_fail(inputQueue, c, encodeOutput); // result_error{} あるいは null を意味する result_finished{}
294  // 5.3.
295  std::string s;
296  from_io_queue_convert(encodeOutput, std::back_inserter(s));
297  for (std::uint8_t byte : s)
298  {
299  // 1.
300  if (spaceAsPlus && byte == '\x20')
301  {
302  output.push_back('\x2B');
303  continue;
304  }
305  // 2.
306  char32_t isomorph = byte;
307  // 3.
308  // 4.
309  if (!set(isomorph)) output.push_back(static_cast<std::uint8_t>(isomorph));
310  // 5.
311  else
312  {
313  std::string pe;
314  percent_encode(byte, std::back_inserter(pe));
315  output.insert(output.end(), pe.begin(), pe.end());
316  }
317  }
318  // 5.4.
319  if (std::holds_alternative<result_error>(potentialError))
320  {
321  std::string_view sv1 = "%26%23";
322  output.insert(output.end(), sv1.begin(), sv1.end());
323  char32_t cp = std::get_if<result_error>(&potentialError)->code_point;
324  serialize_integer(cp, std::back_inserter(output));
325  std::string_view sv2 = "%3B";
326  output.insert(output.end(), sv2.begin(), sv2.end());
327  }
328  }

330  return output;
